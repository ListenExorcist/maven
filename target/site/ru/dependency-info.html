Информация о Зависимостях

Apache Maven

<dependency>
  <groupId>ru.itmo</groupId>
  <artifactId>maven-lesson</artifactId>
  <version>1.0-SNAPSHOT</version>
</dependency>

Apache Buildr

'ru.itmo:maven-lesson:jar:1.0-SNAPSHOT'

Apache Ivy

<dependency org="ru.itmo" name="maven-lesson" rev="1.0-SNAPSHOT">
  <artifact name="maven-lesson" type="jar" />
</dependency>

Groovy Grape

@Grapes(
@Grab(group='ru.itmo', module='maven-lesson', version='1.0-SNAPSHOT')
)

Gradle/Grails

compile 'ru.itmo:maven-lesson:1.0-SNAPSHOT'

Scala SBT

libraryDependencies += "ru.itmo" % "maven-lesson" % "1.0-SNAPSHOT"

Leiningen

[ru.itmo/maven-lesson "1.0-SNAPSHOT"]
